## 详细设计

### listeningRecommend.ts

```ts
import type { ListeningLesson } from '../data/listening'
import { extractErrorWords, type WritingError } from './errorReview'

/** 提取该课 vocabulary 关键词集合 */
export function extractLessonKeywords(lesson: ListeningLesson): Set<string> {
  const set = new Set<string>()
  for (const v of lesson.vocabulary) {
    set.add(v.word.toLowerCase())
  }
  // 也加 blanks 答案 (听写关键词)
  for (const b of lesson.blanks) {
    set.add(b.answer.toLowerCase())
  }
  return set
}

/** 计算该课命中错词数 */
export function calculateLessonScore(
  lesson: ListeningLesson,
  errorWords: Set<string>
): { score: number; matchedWords: string[] } {
  const keywords = extractLessonKeywords(lesson)
  const matched: string[] = []
  for (const w of errorWords) {
    if (keywords.has(w.toLowerCase())) matched.push(w)
  }
  return { score: matched.length, matchedWords: matched }
}

/** 推荐 Top N 课 */
export function recommendLessons(
  lessons: ListeningLesson[],
  errors: WritingError[],
  completedIds: Set<string>,
  topN: number = 3
): Array<{ lesson: ListeningLesson; score: number; matchedWords: string[]; isCompleted: boolean }> {
  if (errors.length === 0) return []

  const errorWords = extractErrorWords(...errors)
  if (errorWords.size === 0) return []

  const scored = lessons.map(lesson => {
    const { score, matchedWords } = calculateLessonScore(lesson, errorWords)
    const isCompleted = completedIds.has(lesson.id)
    // 已完成降权 (但仍可见)
    const adjustedScore = isCompleted ? score * 0.3 : score
    return { lesson, score: adjustedScore, matchedWords, isCompleted }
  })

  // 按 score 降序
  scored.sort((a, b) => b.score - a.score)

  // 只返回 score > 0 的, 取 Top N
  return scored.filter(s => s.score > 0).slice(0, topN)
}
```

### ListenPage 改造

```tsx
// 新 state
const [recommendations, setRecommendations] = useState<Array<...>>([])
const [hasErrors, setHasErrors] = useState(false)

useEffect(() => {
  // 加载错题 → 推荐
  const loadRecs = async () => {
    const errors = await getAllWritingErrors()
    setHasErrors(errors.length > 0)
    if (errors.length > 0) {
      const recs = recommendLessons(LISTENING_LESSONS, errors, completedLessons, 3)
      setRecommendations(recs)
    }
  }
  loadRecs()
}, [completedLessons])

// LessonList 顶部加推荐区
{recommendations.length > 0 && (
  <div className="card bg-gradient-to-br from-amber-50 to-orange-50 dark:from-amber-900/20 dark:to-orange-900/20">
    <h3 className="text-sm font-semibold mb-2">🎯 为你推荐</h3>
    <p className="text-xs text-stone-500 mb-2">根据你的错题推荐</p>
    <div className="space-y-2">
      {recommendations.map(({ lesson, score, matchedWords, isCompleted }) => (
        <button onClick={() => startLesson(lesson)} className="w-full text-left p-2 bg-white dark:bg-stone-800 rounded">
          <div className="flex items-center justify-between">
            <span className="text-sm font-medium">{lesson.title}</span>
            <span className="text-xs">命中 {score} 个错词</span>
          </div>
          {isCompleted && <span className="text-xs text-emerald-500">✓ 已完成</span>}
        </button>
      ))}
    </div>
  </div>
)}

{!hasErrors && (
  <div className="card text-center text-sm text-stone-500 py-4">
    💡 完成几道题后, 这里会出现个性化推荐
  </div>
)}
```

## 风险

- **错题稀疏**: 用户只写 1 篇作文 0 错题, 推荐区不显示
- **命中 0**: 5 课 vocabulary 都不在错词里, 推荐区空 (显示提示)
- **性能**: 每次 completedLessons 变重算, 但 5 课 × N 错词, N<100, 几乎瞬间
- **数据新鲜度**: useEffect 只在 completedLessons 变化时跑, 写作批改后不会自动更新 — 需加 `addEventListener('storage')` 或 `visibilitychange` 重新加载

## 缓解

- 加 visibilitychange 监听 (ListenPage 回到前台时重新计算)
- 错题阈值: errors.length >= 1 才显示推荐 (避免空状态)
- score > 0 才显示 (避免 "命中 0 个错词" 误导)
